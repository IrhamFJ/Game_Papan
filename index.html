const PIECES = { white: { p: '♙', r: '♖', n: '♘', b: '♗', q: '♕', k: '♔', ck: '⚪' }, black: { p: '♟', r: '♜', n: '♞', b: '♝', q: '♛', k: '♚', ck: '🔴' } }; const PIECE_VALUES = { '♟': 10, '♙': 10, '♞': 30, '♘': 30, '♝': 30, '♗': 30, '♜': 50, '♖': 50, '♛': 90, '♕': 90, '♚': 900, '♔': 900, '🔴': 10, '⚪': 10 }; function goToSubMenu(type) { gameType = type; document.getElementById('main-menu').classList.add('hidden'); document.getElementById('sub-menu').classList.remove('hidden'); document.getElementById('mode-subtitle').innerText = type.toUpperCase(); } function backToMain() { document.getElementById('sub-menu').classList.add('hidden'); document.getElementById('main-menu').classList.remove('hidden'); } function launchGame(opp) { opponent = opp; document.getElementById('sub-menu').classList.add('hidden'); document.getElementById('game-screen').classList.remove('hidden'); document.getElementById('current-mode').innerText = `${gameType.toUpperCase()} vs ${opp.toUpperCase()}`; createBoard(); } function createBoard() { const board = document.getElementById('board'); board.innerHTML = ''; for (let r = 0; r < 8; r++) { for (let c = 0; c < 8; c++) { const sq = document.createElement('div'); sq.className = `square ${(r + c) % 2 === 0 ? 'white-sq' : 'black-sq'}`; sq.dataset.row = r; sq.dataset.col = c; let p = "", col = ""; if (gameType === 'chess') { if (r === 1) { p = PIECES.black.p; col = 'black'; } else if (r === 6) { p = PIECES.white.p; col = 'white'; } else if (r === 0 || r === 7) { const side = r === 0 ? 'black' : 'white'; const layout = [PIECES[side].r, PIECES[side].n, PIECES[side].b, PIECES[side].q, PIECES[side].k, PIECES[side].b, PIECES[side].n, PIECES[side].r]; p = layout[c]; col = side; } } else { if ((r + c) % 2 !== 0) { if (r < 3) { p = PIECES.black.ck; col = 'black'; } else if (r > 4) { p = PIECES.white.ck; col = 'white'; } } } if (p) { sq.innerHTML = `${p}`; sq.dataset.color = col; } sq.onclick = () => handleInput(sq); board.appendChild(sq); } } } function handleInput(sq) { if (aiLock) return; if (selectedSq) { if (validateMove(selectedSq, sq)) { processMove(selectedSq, sq); if (opponent === 'ai' && turn === 'black') { aiLock = true; setTimeout(aiBrain, 600); } } else { selectedSq.classList.remove('selected'); selectedSq = null; } } else if (sq.dataset.color === turn) { selectedSq = sq; sq.classList.add('selected'); } } function validateMove(fSq, tSq) { const fR = +fSq.dataset.row, fC = +fSq.dataset.col; const tR = +tSq.dataset.row, tC = +tSq.dataset.col; const dr = Math.abs(tR - fR), dc = Math.abs(tC - fC); const p = fSq.innerText; if (tSq.dataset.color === fSq.dataset.color) return false; if (gameType === 'chess') { if (p === '♘' || p === '♞') return (dr === 2 && dc === 1) || (dr === 1 && dc === 2); if (p === '♗' || p === '♝') return dr === dc && pathClear(fR, fC, tR, tC); if (p === '♖' || p === '♜') return (fR === tR || fC === tC) && pathClear(fR, fC, tR, tC); if (p === '♕' || p === '♛') return (dr === dc || fR === tR || fC === tC) && pathClear(fR, fC, tR, tC); if (p === '♔' || p === '♚') return dr <= 1 && dc <= 1; if (p === '♙' || p === '♟') { const d = fSq.dataset.color === 'white' ? -1 : 1; if (fC === tC && tR === fR + d && !tSq.innerText) return true; // Maju 1 if (fC === tC && fR === (d === 1 ? 1 : 6) && tR === fR + 2*d && !tSq.innerText && pathClear(fR, fC, tR, tC)) return true; // Maju 2 if (dc === 1 && tR === fR + d && tSq.innerText) return true; // Makan diagonal return false; } } else { // Checkers const d = fSq.dataset.color === 'white' ? -1 : 1; if (dc === 1 && tR === fR + d && !tSq.innerText) return true; if (dc === 2 && tR === fR + 2*d) { const mR = (fR + tR) / 2, mC = (fC + tC) / 2; const m = document.querySelector(`[data-row="${mR}"][data-col="${mC}"]`); if (m.dataset.color && m.dataset.color !== fSq.dataset.color) { m.innerHTML = ''; delete m.dataset.color; return true; } } } return false; } function pathClear(fR, fC, tR, tC) { const sR = tR > fR ? 1 : (tR < fR ? -1 : 0), sC = tC > fC ? 1 : (tC < fC ? -1 : 0); let cR = fR + sR, cC = fC + sC; while (cR !== tR || cC !== tC) { if (document.querySelector(`[data-row="${cR}"][data-col="${cC}"]`).innerText) return false; cR += sR; cC += sC; } return true; } function processMove(f, t) { const captured = t.innerText; t.innerHTML = f.innerHTML; t.dataset.color = f.dataset.color; f.innerHTML = ''; delete f.dataset.color; // CEK RAJA MATI if (captured.includes('♔') || captured.includes('♚') || (gameType === 'checkers' && !document.querySelector(`[data-color="${turn === 'white' ? 'black' : 'white'}"]`))) { setTimeout(() => { alert(`SKAKMAT / GAME OVER! PEMENANG: ${turn.toUpperCase()}`); location.reload(); }, 100); return; } // PROMOSI const r = +t.dataset.row; if (gameType === 'chess') { if (t.innerText === PIECES.white.p && r === 0) t.innerHTML = `${PIECES.white.q}`; if (t.innerText === PIECES.black.p && r === 7) t.innerHTML = `${PIECES.black.q}`; } if (selectedSq) selectedSq.classList.remove('selected'); selectedSq = null; turn = (turn === 'white') ? 'black' : 'white'; document.getElementById('turn-label').innerText = turn.toUpperCase(); } function aiBrain() { let moves = []; const all = Array.from(document.querySelectorAll('.square')); const blackPieces = all.filter(s => s.dataset.color === 'black'); blackPieces.forEach(f => { all.forEach(t => { if (validateMove(f, t)) { let score = Math.random() * 2; // Biar gerakan gak monoton const targetPiece = t.innerText; // Prioritas 1: Makan bidak lawan berdasarkan nilai if (targetPiece) { score += PIECE_VALUES[targetPiece] || 0; } // Prioritas 2: Kuasai Tengah const tR = +t.dataset.row, tC = +t.dataset.col; if (tR >= 3 && tR <= 4 && tC >= 3 && tC <= 4) score += 3; moves.push({f, t, score}); } }); }); if (moves.length) { moves.sort((a,b) => b.score - a.score); processMove(moves[0].f, moves[0].t); } aiLock = false; }
